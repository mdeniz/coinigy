def initialize(options = {})
  @headers = {
    'Content-Type': 'application/json',
    'X-API-KEY': options[:key] || '',
    'X-API-SECRET': options[:secret] || ''
  }
  @root = options[:root] || 'https://api.coinigy.com/api/v1'
  @last_request_timestamp = Time.now
end
